program {
  int i;
  for ( i = 1; i <= 5; i = i + 1 ) {
    if ( i % 2 == 0 ) continue;
    write(i);
  }
}

 2:  LOAD 0            // загрузить i

 5:  JMP_FALSE 20      // если i > 5, прыгнуть к выходу из цикла (20 шаг)  

// тело цикла
 6:  LOAD 0            // загрузить i

 8:  MOD               // вычислить i % 2  
 9:  PUSH_INT 0        // загрузить 0  
10:  EQ                // проверить (i % 2 == 0)  
11:  JMP_FALSE 13      // если остаток не 0 (т.е. i нечётно), перейти на 13  
12:  JMP 15            // иначе (i чётно) — continue: прыжок на 15  

// печать i (только для нечётных)
13:  LOAD 0            // загрузить i  
14:  WRITE 1           // напечатать 1 аргумент  

// шаг цикла (увеличение i на 1)
15:  LOAD 0            // загрузить i

17:  ADD               // i + 1
